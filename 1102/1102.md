```
範例2(BMI)
BEGIN
DECLARE BMI int;
SET BMI=weight/(height*height);

IF BMI>24 THEN
SELECT CONCAT('Your BMI is',BMI,'over heavy!'); #CONCAT合併字串
ELSEIF BMI>18 THEN
SELECT CONCAT('Your BMI is',BMI,'Normal');
ELSE
SELECT CONCAT('Your BMI is',BMI,'Skinny');
END IF;
END
```
